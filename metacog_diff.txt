    def _compute_kl_divergence(self, p: np.ndarray, q: np.ndarray) -> float:
        """Kullback-Leibler divergence between two distributions."""
        p = np.clip(p, 1e-12, 1.0)
        q = np.clip(q, 1e-12, 1.0)
        return float(np.sum(p * np.log(p / q)))
